69 struct ModResult {
70  int res;
71  ModResult() : res(0) {}
72  explicit ModResult(int r) : res(r) {}
73  inline bool operator==(const ModResult& r) const
74  {
75  return res == r.res;
76  }
77  inline bool operator!=(const ModResult& r) const
78  {
79  return res != r.res;
80  }
81  inline bool operator!() const
82  {
83  return !res;
84  }
85  inline bool check(bool def) const
86  {
87  return (res == 1 || (res == 0 && def));
88  }
92  inline ModResult operator+(const ModResult& r) const
93  {
94  if (res == r.res || r.res == 0)
95  return *this;
96  if (res == 0)
97  return r;
98  // they are different, and neither is passthru
99  return MOD_RES_ALLOW;
100  }
